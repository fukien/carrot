336 inline CString UpperCase(const CString& AString)
337 {
338  CString result; result.SetLength(AString.GetLength());
339  for (int i = 0, n = AString.GetLength(); i < n; i++)
340  {
341  char c = AString[i];
342  result[i] = islower(c) ? toupper(c) : c;
343  }
344  return result;
345 }
